------------------------------
basics
------------------------------
clone a repo
    git clone uri

checkout branch from repo:
    (after clone)
    display all branches
        git branch -a
    checkout branch from repo
        git checkout 'branchname' (from above command)
    checkout [new branch] (not in repo)
        git branch 'newbranch'
        git checkout 'newbranch'
    checkout file
        git checkout [branchname] -- [path-to-filename]



reset:
    show hashes?? string or whatever representing a commit
        git reflog
    reset to that commit
        git reset --hard 'hash/string_thing'
        NOTES:
            --hard = files get reset too
            if no --hard, just unstages files


pull changes from repo (fetch & merge)
    git pull [remote] [local]
    
push changes to repo
    git push [remote] [local]

tags:
    creation:
        git tag -a [tagname] -m 'tag message/description'
    display:
        git tag -n
    pull tags:
        git fetch --tags
    push tags:
        git push origin [tagname]
