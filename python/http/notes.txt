		# poll_interval: poll for shutdown
		self.__is_shut_down.clear()
		try:
			while not self.__shutdown_request:

	try:
		self.finish_request(req, addr)
	except Exception:
		self.handle_error(req, addr)
	finally:
		self.shutdown_request(req)

def process_request(self, req, addr):
	if self.block_on_close:
		modify _threads to a list of threads-to-be-joined

	def parse_request()
		# Return successful or not
		# On Error, will self.send_error()
		parse the first line of http the
			CMD path HTTP/version
		set close_connection based on the parsed http version, requests, settings, etc
		self.headers = http.client.parse_headers(self.rfile, self.MessageClass)

	def handle_one_request():
		try:
			read the raw request line
			parse_request()
			call do_command, where command is from the parsed request line.
			self.wfile.fush()
		except TimeoutError:
			log
			self.close_connection = True

	def handle():
		self.close_connection = True
		self.handle_one_request()
		while not self.close_connection:
			self.handle_one_request()
